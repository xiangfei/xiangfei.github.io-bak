    country = 'china'

        self.name = name

@classmethod
def run(cls):
    print('%s在奔跑' % cls.country)

Person.run = run
Person.run()

#instance method

import types
class Person(object):
    def __init__(self,name):
        self.name = name

def run(self):
    print('%s在奔跑' % self.name)

p1 = Person('p1')
p1.run = types.MethodType(run,p1)

p1.run()
